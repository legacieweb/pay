1 Quick Integration

Copy and paste this script into your website. You can dynamically set the price and description based on your product data.


<!-- IyonicPay Integration -->
<script src="https://pay.iyonicorp.com/sdk.js"></script>

<!-- Example: A dynamic payment function -->
<script>
function payForProduct(productPrice, productName) {
  IyonicPay.pay({
    username: '...',
    amount: productPrice,
    description: productName,
    onSuccess: (ref) => {
      alert('Payment successful! Reference: ' + ref);
      // Add your logic here (e.g., redirect or update database)
    },
    onCancel: () => alert('Payment cancelled')
  });
}
</script>

<!-- Button Usage -->
<button onclick="payForProduct(25.00, 'Premium Subscription')" style="background:#6366f1; color:white; border:none; padding:12px 24px; border-radius:8px; cursor:pointer; font-weight:bold;">
  Buy Now with IyonicPay
</button>
